%% high level macro set

% graphics in text, #1 filename=label, #2 caption
\newcommand{\textgraphics}[2]{%
\begin{figure}[htbp]%
	\sidecaption{#2\label{#1}}[-16pt]%
	\includegraphics[width=\textwidth]{#1}
	\vspace{1\baselineskip}
\end{figure}%
}

%graphics in the margin, #1 filename=label, #2 caption, #3 offset
\newcommand{\sidegraphics}[3]{%
\begin{sidefigure}[#3]%
	\includegraphics[width=\marginparwidth]{#1}%
	\vspace{2pt}%
	\caption{#2\label{#1}}%
\end{sidefigure}%
}

% table in the margin, #1 label, #2 caption, #3 offset, #4 code
\newcommand{\sidechart}[4]{%
\begin{sidetable}[#3]%
	#4
	 \vspace{2pt}
	\caption{#2\label{#1}}%
\end{sidetable}%
}

% quoted text
\newcommand{\quotetext}{\textit}
